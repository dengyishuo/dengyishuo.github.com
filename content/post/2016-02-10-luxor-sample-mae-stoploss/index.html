require(‘blotter’)

data(‘luxor-p066’, package=‘quantstrat’, envir=.blotter)

currency(c(‘GBP’, ‘USD’)) exchange_rate(c(‘GBPUSD’), tick_size=0.0001)

chart.ME( Portfolio=‘luxor’, Symbol=‘GBPUSD’, type=‘MAE’, scale=‘cash’ )
